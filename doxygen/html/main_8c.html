Die Funktion gibt eine gewisse anzahl der Letzten Eingetragenen URLs aus.

Parameters:
sessionDie IRC Session.
nameDer Benutzen der die URLs wissen will.
counterDie Anzahl wieviele URLs ausgegeben werden. 
{
        sqlite3_stmt *vm;
        sqlite3_prepare(sqlitedb, "SELECT * FROM urls ORDER BY id DESC", -1, &vm, NULL);
        
        int i = 0;      

        //printf("ID:\tnick\tchannel\turl\n");

        while (sqlite3_step(vm) != SQLITE_DONE)
        {
                char tmp[1200];
                sprintf(tmp,"%s (%s - %s)",sqlite3_column_text(vm, 3),sqlite3_column_text(vm, 1),sqlite3_column_text(vm, 2));
                irc_cmd_msg(session,name,tmp);
                i++;
                if(i==counter)
                        break;
        }
        sqlite3_finalize(vm);
        
}
